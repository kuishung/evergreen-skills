# Dependencies for render-audit.py.
#
# In production, sale-audit renders HTML via this script and then hands
# the HTML to `anthropic-skills:pdf` for PDF generation. The HTML path
# only needs jinja2 — weasyprint is NOT required for production runs.
#
#   pip install jinja2     # required, always
#   pip install weasyprint # optional, developer convenience only
#
# weasyprint is only used when the script is invoked with --out <path>.pdf
# directly. That path is for local previews; sale-audit's workflow always
# goes via anthropic-skills:pdf so the pipeline runs in sandboxes (e.g.,
# Cowork's scheduled-task environment) where pip install is blocked.

jinja2>=3.0

# Optional — only for the developer-convenience direct .pdf output path:
# weasyprint>=60.0
